function algoliaHookBeforeWidgetInitialization (allWidgetConfiguration) {
    var stockQty = getStockQuantityConfig();

    if (typeof allWidgetConfiguration['numericRefinementList'] === 'undefined') {
        allWidgetConfiguration['numericRefinementList'] = [stockQty];
    } else {
        allWidgetConfiguration['numericRefinementList'].push(stockQty);
    }

    return allWidgetConfiguration;
};

function getStockQuantityConfig() {
    var wrapper = document.getElementById('instant-search-facets-container');

    return {
        container: wrapper.appendChild(createISWidgetContainer('stock_qty')),
        attributeName: 'stock_qty',
        label: 'In Stock',
        options: [
            { name: 'Any' },
            { start: 1, name: 'In Stock' }
        ],
        cssClasses: {
            root: ['facet']
        },
        templates: {
            header: '<div class="name">Stock Availability</div>'
        }
    };
}

        container: '#instant-search-results-container',
        templates: {

        },
        transformData: {

    };
}
